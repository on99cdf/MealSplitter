    <title>食飯計數助手 - 同事分賬</title>

    <!-- 1. 載入 Tailwind CSS -->

    <script src="https://cdn.tailwindcss.com"></script>

    <!-- 2. 載入 React & Babel -->

    <script src="https://unpkg.com/react@18/umd/react.production.min.js"></script>

    <script src="https://unpkg.com/react-dom@18/umd/react-dom.production.min.js"></script>

    <script src="https://unpkg.com/@babel/standalone/babel.min.js"></script>

    <!-- 3. 載入 Firebase 兼容版 (Compat) - 最穩定 -->

    <script src="https://www.gstatic.com/firebasejs/10.7.1/firebase-app-compat.js"></script>

    <script src="https://www.gstatic.com/firebasejs/10.7.1/firebase-auth-compat.js"></script>

    <script src="https://www.gstatic.com/firebasejs/10.7.1/firebase-firestore-compat.js"></script>

    <style>

        @keyframes spin { to { transform: rotate(360deg); } }

        .spinner { animation: spin 1s linear infinite; border-top-color: transparent; }

        body { -webkit-tap-highlight-color: transparent; font-family: -apple-system, BlinkMacSystemFont, "Segoe UI", Roboto, Helvetica, Arial, sans-serif; }

        .modal-bg { background-color: rgba(0, 0, 0, 0.4); backdrop-filter: blur(4px); }

        .safe-pb { padding-bottom: env(safe-area-inset-bottom, 24px); }

<body class="bg-gray-50 antialiased text-slate-900">

    <div id="root"></div>

    <script type="text/babel">

        // --- 核心配置 ---

        // 當你在 Tiiny.host 或其他地方發佈時，這段配置會自動從環境變量讀取

        // 如果是在本地運行，請確保你已經將正確的 config 填入

        const firebaseConfig = (() => {

            try { return JSON.parse(window.__firebase_config || '{}'); } 

            catch(e) { return {}; }

        })();

        

        const appId = window.__app_id || 'lunch-group-v1';

        const apiKey = ""; // Gemini API Key

        // 初始化 Firebase

        if (firebaseConfig.apiKey) {

            if (!firebase.apps.length) firebase.initializeApp(firebaseConfig);

        }

        

        const db = firebase.firestore ? firebase.firestore() : null;

        const auth = firebase.auth ? firebase.auth() : null;

        // --- 內建圖標 ---

        const Icons = {

            Camera: () => <svg width="24" height="24" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round"><path d="M23 19a2 2 0 0 1-2 2H3a2 2 0 0 1-2-2V8a2 2 0 0 1 2-2h4l2-3h6l2 3h4a2 2 0 0 1 2 2z"></path><circle cx="12" cy="13" r="4"></circle></svg>,

            Users: () => <svg width="20" height="20" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round"><path d="M17 21v-2a4 4 0 0 0-4-4H5a4 4 0 0 0-4 4v2"></path><circle cx="9" cy="7" r="4"></circle></svg>,

            Receipt: () => <svg width="20" height="20" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round"><path d="M4 2v20l2-1 2 1 2-1 2 1 2-1 2 1 2-1 2 1V2l-2 1-2-1-2 1-2-1-2 1-2-1-2 1-2-1z"></path></svg>,

            History: () => <svg width="20" height="20" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round"><circle cx="12" cy="12" r="10"></circle><polyline points="12 6 12 12 16 14"></polyline></svg>,

            Trash: () => <svg width="18" height="18" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round"><polyline points="3 6 5 6 21 6"></polyline><path d="M19 6v14a2 2 0 0 1-2 2H7a2 2 0 0 1-2-2V6m3 0V4a2 2 0 0 1 2-2h4a2 2 0 0 1 2 2v2"></path></svg>,

            Check: () => <svg width="16" height="16" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="3" strokeLinecap="round" strokeLinejoin="round"><polyline points="20 6 9 17 4 12"></polyline></svg>,

            X: () => <svg width="20" height="20" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round"><line x1="18" y1="6" x2="6" y2="18"></line><line x1="6" y1="6" x2="18" y2="18"></line></svg>

        };

        const { useState, useEffect, useMemo, useRef } = React;

        const App = () => {

            const [user, setUser] = useState(null);

            const [activeTab, setActiveTab] = useState('dashboard');

            const [members, setMembers] = useState([]);

            const [receipts, setReceipts] = useState([]);

            const [loading, setLoading] = useState(false);

            const [currentReceipt, setCurrentReceipt] = useState(null);

            const [payerId, setPayerId] = useState('');

            const [splitMode, setSplitMode] = useState('itemized');

            const [equalMembers, setEqualMembers] = useState([]);

            const [modal, setModal] = useState({ type: null, data: null });

            const [newName, setNewName] = useState('');

            const fileInputRef = useRef(null);

            // 1. 登入與基礎檢查

            useEffect(() => {

                if (!auth) return;

                auth.signInAnonymously().catch(err => console.error("Firebase Login Error", err));

                return auth.onAuthStateChanged(setUser);

            }, []);

            // 2. 監聽 Firestore (即時同步)

            useEffect(() => {

                if (!user || !db) return;

                const baseRef = db.collection('artifacts').doc(appId).collection('public').doc('data');

                

                const unsubM = baseRef.collection('members').onSnapshot(s => {

                    setMembers(s.docs.map(d => ({ id: d.id, ...d.data() })));

                });

                

                const unsubR = baseRef.collection('receipts').onSnapshot(s => {

                    setReceipts(s.docs.map(d => ({ id: d.id, ...d.data() })).sort((a,b) => b.timestamp - a.timestamp));

                });

                

                return () => { unsubM(); unsubR(); };

            }, [user]);

            // 3. 計算誰欠誰

            const settlements = useMemo(() => {

                let debtors = [], creditors = [];

                members.forEach(m => {

                    const b = m.balance || 0;

                    if (b < -0.1) debtors.push({ name: m.name, bal: Math.abs(b) });

                    else if (b > 0.1) creditors.push({ name: m.name, bal: b });

                });

                debtors.sort((a,b) => b.bal - a.bal); creditors.sort((a,b) => b.bal - a.bal);

                const res = []; let d=0, c=0;

                while(d < debtors.length && c < creditors.length) {

                    const amt = Math.min(debtors[d].bal, creditors[c].bal);

                    if(amt > 0.1) res.push({ from: debtors[d].name, to: creditors[c].name, amt });

                    debtors[d].bal -= amt; creditors[c].bal -= amt;

                    if(debtors[d].bal < 0.1) d++; if(creditors[c].bal < 0.1) c++;

                }

                return res;

            }, [members]);

            // 4. OCR 掃描 (AI)

            const handleScan = async (e) => {

                const file = e.target.files[0];

                if (!file) return;

                setLoading(true);

                const reader = new FileReader();

                reader.onloadend = async () => {

                    const base64 = reader.result.split(',')[1];

                    try {

                        const resp = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`, {

                            method: 'POST',

                            headers: { 'Content-Type': 'application/json' },

                            body: JSON.stringify({

                                contents: [{ parts: [{ text: "辨識此收據。只回傳 JSON: {\"items\": [{\"name\": \"品名\", \"price\": 100.0}], \"total\": 總金額}" }, { inlineData: { mimeType: "image/png", data: base64 } }] }],

                                generationConfig: { responseMimeType: "application/json" }

                            })

                        });

                        const data = await resp.json();

                        const json = JSON.parse(data.candidates[0].content.parts[0].text);

                        setCurrentReceipt({

                            items: json.items.map(i => ({ ...i, claim: [] })),

                            total: json.total || json.items.reduce((s,x) => s + x.price, 0),

                            timestamp: Date.now()

                        });

                        setPayerId(''); setSplitMode('itemized'); setActiveTab('split');

                    } catch (err) { console.error("AI 辨識出錯", err); } finally { setLoading(false); }

                };

                reader.readAsDataURL(file);

            };

            // 5. 保存分賬結果

            const saveSplit = async () => {

                if (!payerId) { setModal({ type: 'alert', data: '請先選擇付款人！' }); return; }

                setLoading(true);

                try {

                    const shares = {}; members.forEach(m => shares[m.id] = 0);

                    if (splitMode === 'itemized') {

                        const sub = currentReceipt.items.reduce((s,i) => s + i.price, 0);

                        const ratio = sub === 0 ? 1 : currentReceipt.total / sub;

                        currentReceipt.items.forEach(it => {

                            if (it.claim.length > 0) {

                                const p = (it.price * ratio) / it.claim.length;

                                it.claim.forEach(id => shares[id] += p);

                            }

                        });

                    } else {

                        if (!equalMembers.length) { setModal({type:'alert', data:'請選擇平分成員'}); setLoading(false); return; }

                        const p = currentReceipt.total / equalMembers.length;

                        equalMembers.forEach(id => shares[id] = p);

                    }

                    const batch = db.batch();

                    const baseRef = db.collection('artifacts').doc(appId).collection('public').doc('data');

                    const payer = members.find(m => m.id === payerId);

                    let pBal = (payer.balance || 0) + currentReceipt.total;

                    

                    members.forEach(m => {

                        const s = shares[m.id] || 0;

                        if (m.id === payerId) pBal -= s;

                        else if (s > 0) batch.update(baseRef.collection('members').doc(m.id), { balance: (m.balance || 0) - s });

                    });

                    batch.update(baseRef.collection('members').doc(payerId), { balance: pBal });

                    

                    await baseRef.collection('receipts').add({

                        ...currentReceipt, shares, payerId, splitMode, settled: [], timestamp: Date.now()

                    });

                    await batch.commit();

                    setCurrentReceipt(null); setActiveTab('dashboard');

                } catch (e) { console.error(e); } finally { setLoading(false); }

            };

            // 6. 標記已收錢 (即時連動)

            const togglePaid = async (receipt, mid, amt) => {

                const isPaid = (receipt.settled || []).includes(mid);

                const batch = db.batch();

                const baseRef = db.collection('artifacts').doc(appId).collection('public').doc('data');

                const debtor = members.find(m => m.id === mid);

                const payer = members.find(m => m.id === receipt.payerId);

                const rRef = baseRef.collection('receipts').doc(receipt.id);

                

                if (!isPaid) {

                    batch.update(baseRef.collection('members').doc(mid), { balance: (debtor.balance || 0) + amt });

                    batch.update(baseRef.collection('members').doc(receipt.payerId), { balance: (payer.balance || 0) - amt });

                    batch.update(rRef, { settled: firebase.firestore.FieldValue.arrayUnion(mid) });

                } else {

                    batch.update(baseRef.collection('members').doc(mid), { balance: (debtor.balance || 0) - amt });

                    batch.update(baseRef.collection('members').doc(receipt.payerId), { balance: (payer.balance || 0) + amt });

                    batch.update(rRef, { settled: firebase.firestore.FieldValue.arrayRemove(mid) });

                }

                await batch.commit();

            };

            // --- 防崩潰渲染 ---

            if (!firebaseConfig.apiKey) {

                return (

                    <div className="p-10 text-center space-y-4">

                        <div className="text-rose-500 font-bold text-lg">系統未正確初始化</div>

                        <p className="text-slate-500 text-sm">請確保您是在預覽環境中打開，或已正確填寫 Firebase 配置。</p>

                    </div>

                );

            }

            if (!user) return <div className="h-screen flex items-center justify-center font-black text-indigo-600 animate-pulse">正在登入伺服器...</div>;

            return (

                <div className="max-w-md mx-auto min-h-screen bg-slate-50 flex flex-col relative shadow-2xl">

                    <header className="p-4 bg-white border-b flex justify-between items-center sticky top-0 z-40">

                        <div className="flex items-center gap-2">

                            <div className="bg-indigo-600 text-white p-2 rounded-xl"><Icons.Receipt /></div>

                            <h1 className="font-black text-xl italic tracking-tighter text-slate-800">MEALSPLITTER</h1>

                        </div>

                        <button onClick={() => fileInputRef.current.click()} className="bg-indigo-600 text-white p-2.5 rounded-xl active:scale-95 transition-all">

                            <Icons.Camera />

                        </button>

                        <input type="file" ref={fileInputRef} onChange={handleScan} accept="image/*" className="hidden" />

                    </header>

                    <main className="flex-1 overflow-y-auto p-4 pb-32">

                        {activeTab === 'dashboard' && (

                            <div className="space-y-6">

                                <div className="bg-indigo-600 rounded-[32px] p-7 text-white shadow-xl">

                                    <h2 className="text-3xl font-black mb-6">結餘總覽</h2>

                                    <div className="grid grid-cols-2 gap-4">

                                        <div className="bg-white/10 p-4 rounded-2xl"><p className="text-[10px] opacity-60 font-bold uppercase mb-1">成員</p><p className="text-xl font-bold">{members.length} 人</p></div>

                                        <div className="bg-white/10 p-4 rounded-2xl"><p className="text-[10px] opacity-60 font-bold uppercase mb-1">單據</p><p className="text-xl font-bold">{receipts.length} 張</p></div>

                                    </div>

                                </div>

                                {settlements.length > 0 && (

                                    <div className="bg-emerald-50 border border-emerald-100 rounded-2xl p-5 shadow-sm">

                                        <h3 className="text-xs font-black text-emerald-700 mb-4 uppercase tracking-widest flex items-center gap-1">建議清數方案</h3>

                                        {settlements.map((s, i) => (

                                            <div key={i} className="flex justify-between bg-white p-3.5 rounded-xl mb-2 text-sm shadow-sm border border-emerald-50">

                                                <span className="font-bold text-slate-700">{s.from} ➔ {s.to}</span>

                                                <span className="font-black text-emerald-600">HK${s.amt.toFixed(1)}</span>

                                            </div>

                                        ))}

                                    </div>

                                )}

                                <div className="flex justify-between items-center px-1">

                                    <h3 className="font-black text-slate-800">同事結欠</h3>

                                    <button onClick={() => setModal({ type: 'add' })} className="text-indigo-600 text-xs font-bold bg-indigo-50 px-4 py-2 rounded-xl">+ 新增成員</button>

                                </div>

                                <div className="space-y-3">

                                    {members.map(m => (

                                        <div key={m.id} className="bg-white p-4 rounded-2xl shadow-sm border border-slate-100 flex justify-between items-center">

                                            <div className="flex items-center gap-3">

                                                <div className="w-10 h-10 bg-slate-50 text-indigo-600 rounded-xl flex items-center justify-center font-bold">{m.name?.[0]}</div>

                                                <span className="font-bold text-slate-700">{m.name}</span>

                                            </div>

                                            <div className="flex items-center gap-3">

                                                <span className={`font-mono font-black text-lg ${m.balance < -0.1 ? 'text-rose-500' : m.balance > 0.1 ? 'text-emerald-500' : 'text-slate-300'}`}>

                                                    {m.balance < -0.1 ? `-HK$${Math.abs(m.balance).toFixed(1)}` : m.balance > 0.1 ? `+HK$${m.balance.toFixed(1)}` : 'HK$0'}

                                                </span>

                                                <button onClick={() => setModal({ type: 'delete', data: m })} className="text-slate-200"><Icons.Trash /></button>

                                            </div>

                                        </div>

                                    ))}

                                </div>

                            </div>

                        )}

                        {activeTab === 'split' && currentReceipt && (

                            <div className="space-y-6">

                                <div className="flex justify-between items-center">

                                    <h2 className="text-2xl font-black">分配單據</h2>

                                    <div className="flex bg-gray-200 p-1 rounded-xl">

                                        <button onClick={() => setSplitMode('itemized')} className={`px-4 py-2 rounded-lg text-xs font-bold ${splitMode === 'itemized' ? 'bg-white shadow-sm text-indigo-600' : 'text-slate-500'}`}>按餐點</button>

                                        <button onClick={() => setSplitMode('equal')} className={`px-4 py-2 rounded-lg text-xs font-bold ${splitMode === 'equal' ? 'bg-white shadow-sm text-indigo-600' : 'text-slate-500'}`}>全平分</button>

                                    </div>

                                </div>

                                <div className="bg-indigo-50 p-6 rounded-3xl border border-indigo-100 flex justify-between items-center shadow-inner">

                                    <span className="font-bold text-indigo-400 text-sm uppercase">總金額</span>

                                    <span className="text-3xl font-black text-indigo-600">HK${currentReceipt.total.toFixed(1)}</span>

                                </div>

                                <section>

                                    <p className="text-xs font-bold text-slate-400 mb-3 uppercase tracking-widest px-1">誰墊支了整筆賬單？</p>

                                    <div className="flex flex-wrap gap-2">

                                        {members.map(m => (

                                            <button key={m.id} onClick={() => setPayerId(m.id)} className={`px-4 py-2 rounded-xl text-xs font-bold border-2 transition-all ${payerId === m.id ? 'bg-indigo-600 border-indigo-600 text-white shadow-lg' : 'bg-white border-slate-100 text-slate-500'}`}>{m.name}</button>
